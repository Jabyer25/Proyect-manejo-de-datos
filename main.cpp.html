#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<fstream>
#include<cstdlib>
using namespace std;

char nombre[250];
string nombreArchivo;
string frase;
ofstream archivo;

void indentificador(){
    cout<<"Digite el nombre del archivo: ";
    cin>>nombre;
}
int opcion;

void escribir(){

    char band;
    cout<<"Digite el nombre del archivo: ";
    cin>>nombreArchivo;
    archivo.open(nombreArchivo.c_str(),ios::app);

    if(archivo.fail()){
        cout<<"No se pudo abrir el archivo";
        exit(1);
    }

    do{
        fflush(stdin);

        cout<<"\nDigite el texto del archivo: ";
        getline(cin,frase);

        archivo<<frase<<endl;

        cout<<"\nDesea agregar mas texto al archivo? (s/n)";
        cin>>band;
    }while((band == 's') || (band == 'S'));

    archivo.close();

    }

    void comprimir(){
        indentificador();
        sprintf(nombre, "7z a COMPRIMIDO.7Z", nombre);
        system(nombre);
}

void eliminar(){
    indentificador();
    remove(nombre);
}

int main(){
    cout<<"Opcion 1 abrir un archivo y agregarle texto\n";
    cout<<"Opcion 2 comprimir archivo\n";
    cout<<"Opcion 3 eliminar archivo\n";
    cin>>opcion;

    switch (opcion) {
        case 1:
            escribir();
            break;
        case 2:
            comprimir();
            break;
        case 3:
            eliminar();
            break;
    }

}
